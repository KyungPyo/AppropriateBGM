CREATE TABLE IF NOT EXISTS Category (
  category_id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  category_name VARCHAR(20) NOT NULL);

CREATE TABLE IF NOT EXISTS BGMList (
  bgm_name VARCHAR(255) NOT NULL,
  bgm_path VARCHAR(255) PRIMARY KEY NOT NULL,
  innerfile INTEGER NOT NULL DEFAULT 0,
  category_id INTEGER NOT NULL DEFAULT 2,
  CONSTRAINT fk_BGMList_Category FOREIGN KEY (category_id) REFERENCES Category (category_id));

CREATE TABLE IF NOT EXISTS Favorite (
  favorite_id INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
  bgm_path VARCHAR(255) NULL,
  CONSTRAINT fk_Favorite_BGMList1 FOREIGN KEY (bgm_path) REFERENCES BGMList (bgm_path));
